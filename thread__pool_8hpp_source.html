46 class ThreadPool {
47  public:
48  ThreadPool(size_t);
49  template <class F, class... Args>
50  auto enqueue(F&& f, Args&&... args) -> std::future<typename std::result_of<F(Args...)>::type>;
51  ~ThreadPool();

53  private:
54  // need to keep track of threads so we can join them
55  std::vector<std::thread> workers;
56  // the task queue
57  std::queue<std::function<void()>> tasks;
58 
59  // synchronization
60  std::mutex queue_mutex;
61  std::condition_variable condition;
62  bool stop;
63 };

65 // the constructor just launches some amount of workers
66 inline ThreadPool::ThreadPool(size_t threads) : stop(false) {
67  for (size_t i = 0; i < threads; ++i)
68  workers.emplace_back([this] {
69  for (;;) {
70  std::function<void()> task;
71 
72  {
73  std::unique_lock<std::mutex> lock(this->queue_mutex);
74  this->condition.wait(lock, [this] { return this->stop || !this->tasks.empty(); });
75  if (this->stop && this->tasks.empty()) return;
76  task = std::move(this->tasks.front());
77  this->tasks.pop();
78  }
79 
80  task();
81  }
82  });
83 }

85 // add new work item to the pool
86 template <class F, class... Args>
87 auto ThreadPool::enqueue(F&& f, Args&&... args) -> std::future<typename std::result_of<F(Args...)>::type> {
88  using return_type = typename std::result_of<F(Args...)>::type;
89 
90  auto task =
91  std::make_shared<std::packaged_task<return_type()>>(std::bind(std::forward<F>(f), std::forward<Args>(args)...));
92 
93  std::future<return_type> res = task->get_future();
94  {
95  std::unique_lock<std::mutex> lock(queue_mutex);
96 
97  // don't allow enqueueing after stopping the pool
98  if (stop) throw std::runtime_error("enqueue on stopped ThreadPool");
99 
100  tasks.emplace([task]() { (*task)(); });
101  }
102  condition.notify_one();
103  return res;
104 }

106 // the destructor joins all threads
107 inline ThreadPool::~ThreadPool() {
108  {
109  std::unique_lock<std::mutex> lock(queue_mutex);
110  stop = true;
111  }
112  condition.notify_all();
113  for (std::thread& worker : workers) worker.join();
114 }
